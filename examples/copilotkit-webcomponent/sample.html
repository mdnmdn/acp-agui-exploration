Usage Instructions

<!-- 1. Import the web component -->
<script type="module" src="./dist/index.js"></script>

<!-- 2. Use the custom element -->
<acp-copilotkit
  url="http://localhost:4000/copilotkit"
  apikey="your-public-key"
  mode="sidebar"
  labels-title="My Assistant"
  labels-initial="Hello!"
  labels-placeholder="Type a message..."
  default-open="true"
></acp-copilotkit>

<!-- 3. Listen for events -->
<script>
  const el = document.querySelector('acp-copilotkit');
  el.addEventListener('copilotkit:message-sent', (e) => {
    console.log('User sent:', e.detail.message);
  });
  el.addEventListener('copilotkit:open', () => console.log('Chat opened'));
  el.addEventListener('copilotkit:close', () => console.log('Chat closed'));
</script>
